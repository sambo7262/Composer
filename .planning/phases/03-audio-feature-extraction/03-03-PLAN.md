    IDLE = "idle"
    RUNNING = "running"
    PAUSED = "paused"
    COMPLETED = "completed"
    FAILED = "failed"

class AnalysisStatus:
    state: AnalysisStateEnum
    total_tracks: int
    analyzed_tracks: int
    failed_tracks: int
    current_track: str
    avg_seconds_per_track: float
    errors: list[dict]  # [{track: str, error: str}, ...]
    # property: eta_display -> str

          - If un_analyzed_count == 0 and analyzed_count > 0: "All {{ analyzed_count }} tracks analyzed" (no button needed)
